## 前导零

看范围给的是字符串 注意处理

```
#include<bits/stdc++.h> 
using namespace std;
int num[210];
int main(){
	bool preZ=true;
	int len=0;
	//读取字符串 
	char tc='0';
	while(cin>>tc){
		//前导零处理 
		if(preZ&&tc=='0')
			continue;	
		preZ=false;
		num[len++]=tc-'0';
	}	 
	for(int i=0;i<len;i++){
		int pn=len-1-i;//对应数  
		if(num[i]<=num[pn]){
			num[pn]=num[i];
		}else{//对应数增加 向前借位 
			
			int fn=pn-1;//被借位的数
			//处理被借位的数为零的情况 一直向前借位 
			while(num[fn]==0)
				num[fn--]=9;
			num[fn]--;
      num[pn]=num[i];
			if(fn==0&&num[fn]==0){//最前面一位被借成0 则pn到首位之间全是0
				//又因为 0<=i<pn(i==pn时不可能有num[pn]<num[i]) 且 num[pn]<num[i]
				//则只有num[0]=1 其余位置均为0的情况 
				//此时除头尾两位 其余均为9
				num[0]=9;
				len--;
				break;
			}
		} 
	}
	for(int i=0;i<len;i++)
		cout<<num[i];
	return 0;
}
```
